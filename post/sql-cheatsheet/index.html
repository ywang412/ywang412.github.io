rank scores

SELECT Score, 
(SELECT COUNT(DISTINCT Score) FROM Scores WHERE Score >= s.Score) Rank 
FROM Scores s ORDER BY Score DESC;

SELECT s.Score, COUNT(DISTINCT t.Score) Rank
FROM Scores s JOIN Scores t ON s.Score <= t.Score
GROUP BY s.Id ORDER BY s.Score DESC;

Exchange seat

select
if(id < (select count(*) from seat), if(id mod 2=0, id-1, id+1), if(id mod 2=0, id-1, id)) as id, student
from seat
order by id asc;

SELECT (CASE 
    WHEN mod(id, 2) != 0 and records != id THEN id + 1
    WHEN mod(id, 2) != 0 and records = id THEN id
    ELSE id - 1
END) AS id, student
FROM seat, (select count(*) as records from seat) as seat_records
ORDER BY id asc;

Friend Requests I

select
round(
    ifnull(
    (select count(*) from (select distinct requester_id, accepter_id from request_accepted) as A)
    /
    (select count(*) from (select distinct sender_id, send_to_id from friend_request) as B),
    0)
, 2) as accept_rate;

Friend Requests II

select id, count(*) as num
   from
   (
        select requester_id as id from request_accepted
        union all   -- union all will take duplicates
        select accepter_id from request_accepted
    ) as tbl1
   group by id
order by num desc
limit 1
;

SELECT id1 AS id, COUNT() AS num
FROM (
SELECT requester_id AS id1, accepter_id AS id2 FROM request_accepted
UNION
SELECT accepter_id AS id1 , requester_id AS id2 FROM request_accepted
) a
GROUP BY 1
ORDER BY COUNT() DESC
LIMIT 1
